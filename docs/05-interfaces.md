### 1.1 `ITableEngine`

**File:** `packages/core/src/engine/ITableEngine.ts`

```ts
import type { Operation, Cell, TableState, TableSchema } from "../models";

export interface ITableEngine {
  /**
   * Applies a single operation to the table state.
   * Guarantees determinism and idempotence (same op.id applied twice = no-op on second call).
   * Has no side effects: no IO, no network calls, no random values.
   */
  apply(op: Operation): void;

  /**
   * Applies a sequence of operations in order.
   * Equivalent to calling apply() for each op, but may be optimized for batch scenarios.
   */
  applyBatch(ops: Operation[]): void;

  /**
   * Returns the Cell at the given position.
   * If the cell does not exist, returns a default cell: { value: null } or column default.
   * Never throws for missing cells.
   */
  getCell(rowId: string, colId: string): Cell;

  /**
   * Returns the current table state snapshot (schema + rows + rowOrder).
   * The returned state is a reference; do not mutate directly.
   */
  getState(): TableState;

  /**
   * Returns the current schema.
   */
  getSchema(): TableSchema;

  /**
   * Returns the current ordered list of row IDs.
   */
  getRowOrder(): string[];
}
```

## 2. Package: `operations`

### 2.1 `IOperationManager`

**File:** `packages/operations/src/IOperationManager.ts`

```ts
import type { Operation, TargetDescriptor } from "@advanced-datatable/core";
import type { OperationError, OperationRecord } from "./types";

export interface IOperationManager {
  /**
   * Registers an operation as pending, applies it optimistically to the engine,
   * and enqueues it for dispatch to the backend.
   * - If op.id has already been applied, this is a no-op.
   */
  apply(op: Operation): void;

  /**
   * Applies a batch of operations sequentially.
   * Equivalent to calling apply() for each op in order.
   */
  applyBatch(ops: Operation[]): void;

  /**
   * Marks a pending/applying operation as confirmed.
   * Removes it from the pending tracking index.
   * Called when backend returns ack for opId.
   */
  confirm(opId: string): void;

  /**
   * Marks an operation as failed.
   * Removes it from the pending tracking index.
   * Optionally triggers a revert in the engine.
   * Called when backend returns error for opId.
   */
  fail(opId: string, error?: OperationError): void;

  /**
   * Returns all operation IDs currently pending/applying for the given target.
   * Used by the store to derive loading state for UI cells/rows/columns.
   * Returns empty array if no pending operations for target.
   */
  getPendingByTarget(target: TargetDescriptor): string[];

  /**
   * Returns the current record for a given operation ID.
   * Returns undefined if the operation is not tracked.
   */
  getRecord(opId: string): OperationRecord | undefined;

  /**
   * Returns all currently pending/applying operations.
   */
  getAllPending(): OperationRecord[];
}
```

**File:** `packages/api-client/src/IOperationTransport.ts`

```ts
import type { Operation } from "@advanced-datatable/core";
import type { SubmitOperationsResponse, TableLoadResponse } from "./types";

export interface IOperationTransport {
  /**
   * Sends a batch of operations to the backend.
   * Returns the backend's response: acks, errors, and optional server-initiated ops.
   * Rejects with a network/transport error if the request fails entirely.
   */
  send(ops: Operation[]): Promise<SubmitOperationsResponse>;

  /**
   * Loads the initial table state from the backend.
   */
  loadTable(): Promise<TableLoadResponse>;

  /**
   * (Optional) Subscribes to server-push operations (WebSocket).
   * The callback is invoked with new operations received from the server.
   * Returns an unsubscribe function.
   */
  subscribe?: (callback: (ops: Operation[]) => void) => () => void;
}
```
